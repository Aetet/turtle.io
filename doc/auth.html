src/auth.js

Method auth

Determines what authentication is valid (if any), and applies it to the request

Parameters:

  • req must be an Object.
    (Request Object)

  • res must be an Object.
    (Response Object)

  • host must be a String.
    (Virtual host)

  • callback must be a Function.
    (Function to execute after applying optional authenication wrapper)

Returns an Object
(TurtleIO instance)

TurtleIO.prototype.auth = function ( req, res, host, callback ) {

No authentication

if ( !this.config.auth || ( this.config.auth.basic && !this.config.auth.basic[host] ) ) { callback(); }

Basic authentication

else if ( this.config.auth.basic && this.config.auth.basic[host] ) { if ( !this.config.auth.basic[host].auth ) { this.config.auth.basic[host].auth = http_auth( this.config.auth.basic[host] ); } this.config.auth.basic[host].auth.apply( req, res, callback ); } return this; };
